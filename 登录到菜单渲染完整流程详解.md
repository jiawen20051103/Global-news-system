## 一、完整数据流和组件生命周期

### 🎯 整体流程图

```
用户输入账号密码
    ↓
Login.jsx 提交表单
    ↓
请求后端验证 (GET /users?username=xx&password=yy&_embed=role)
    ↓
后端返回用户数据（包含 role 对象）
    ↓
保存到 localStorage.token
    ↓
跳转到 /home (或原页面)
    ↓
IndexRouter 组件挂载
    ↓
获取路由配置 (GET /rights + GET /children)
    ↓
解析 token 获取用户权限
    ↓
动态生成路由 (根据权限过滤)
    ↓
渲染 NewsSandBox
    ↓
SideMenu 组件挂载
    ↓
获取用户权限 (从 localStorage.token)
    ↓
获取菜单配置 (GET /rights?_embed=children)
    ↓
根据权限过滤菜单
    ↓
渲染菜单树
```

### 📝 详细步骤解析

#### **阶段1：用户登录 (Login.jsx)**

**组件生命周期：**
```javascript
Login 组件挂载
  ↓
useEffect(() => { initParticlesEngine() }, [])  // 初始化粒子背景
  ↓
用户填写表单并点击"登录"按钮
  ↓
Form.onFinish 触发 → onFinish(values)
```

**数据流：**
```javascript
// 1. 用户输入数据
values = {
  username: "admin",
  password: "123456"
}

// 2. 发起登录请求
request.get(`/users?username=${values.username}&password=${values.password}&roleState=true&_embed=role`)
```

**后端处理（json-server）：**
- 从 `db.json` 的 `users` 数组中查找匹配的用户
- 使用 `_embed=role` 参数，自动关联 `roles` 表
- 返回完整的用户对象，包含 `role` 对象

**返回数据结构：**
```json
{
  "id": "1",
  "username": "admin",
  "password": "123456",
  "region": "全球",
  "roleId": "1",
  "roleState": true,
  "role": {
    "id": "1",
    "roleName": "超级管理员",
    "roleType": 1,
    "rights": [
      "/home",
      "/user-manage/list",
      "/news-manage/add",
      // ... 更多权限
    ]
  }
}
```

**保存 token：**
```javascript
// 3. 保存到 localStorage
localStorage.setItem('token', JSON.stringify(res.data[0]))
// 此时 localStorage.token 存储的是整个用户对象的 JSON 字符串
```

**路由跳转：**
```javascript
// 4. 跳转到目标页面
const from = location.state?.from?.pathname || '/home'
navigate(from, { replace: true })
```

```javascript
let role = { rights: [] }
if (isLogin && token && token.role) {
  // 已登录：使用用户的实际权限
```

**getTokenInfo() 执行过程：**
```javascript
// src/util/getTokenInfo.js
const getTokenInfo = () => {
  const tokenStr = localStorage.getItem('token');
  // tokenStr = '{"id":"1","username":"admin",...,"role":{...}}'
  
  if (!tokenStr) return {};
  
  try {
    return JSON.parse(tokenStr);
    // 返回完整的用户对象，包含 role 对象
  } catch (err) {
    console.error('token 解析失败', err);
    localStorage.removeItem('token');
    return {};
  }
};
```

**步骤3：权限检查函数**
```javascript
// IndexRouter.jsx 第87-97行
const checkRoute = (item) => {
  // 检查路由是否存在且允许访问
  return LocalRouterMap[item.key] && (item.pagepermisson || item.routepermisson)
}

const checkUserPermission = (item) => {
  if (!Array.isArray(role.rights) || role.rights.length === 0) {
    // 没有权限信息时，默认放行（避免页面空白）
    return true
  }
  // 检查用户的 rights 数组中是否包含该路由的 key
  return role.rights.includes(item.key)
}
```

**步骤4：动态生成路由**
```javascript
// IndexRouter.jsx 第119-127行
{BackRouterList.map((item) => {
  const Component = LocalRouterMap[item.key]  // 获取对应的组件
  
  // 双重检查：路由存在 + 用户有权限
  if (checkRoute(item) && checkUserPermission(item)) {
    const routePath = item.key.startsWith('/') ? item.key.slice(1) : item.key
    return <Route path={routePath} key={item.key} element={<Component />} />
  }
  return null
})}
```

```javascript
SideMenu 组件挂载

useEffect(() => {

  ↓

  // 当 rights 变化时，重新获取菜单配置
  request.get('/rights?_embed=children')
    .then(res => {
      const formattedMenu = transformMenuData(res.data)
```

**详细数据流：**

**步骤1：获取用户权限**
```javascript
// SideMenu.jsx 第75-111行
useEffect(() => {
  const getRights = () => {
    try {
      const isLogin = checkLogin();
      if (isLogin) {
        const token = localStorage.getItem('token');
        if (token) {
          const { role } = JSON.parse(token);
          setRights(role?.rights || []);  // 提取权限数组
        } else {
          setRights([]);
        }
      } else {
        // 未登录用户使用区域编辑的权限（只读）
        setRights(getGuestRights());
      }
    } catch (error) {
      console.error('解析token失败:', error);
      setRights(getGuestRights());
    }
  };

  getRights();

  // 监听 localStorage 变化（跨标签页同步）
  const handleStorageChange = (e) => {
    if (e.key === 'token') {
      getRights();  // token 变化时重新获取权限
    }
  };
  window.addEventListener('storage', handleStorageChange);

  return () => {
    window.removeEventListener('storage', handleStorageChange);
  };
}, []);
```

**步骤3：权限过滤和菜单转换**
```javascript
// SideMenu.jsx 第34-72行
const transformMenuData = useCallback((items) => {
  if (!Array.isArray(items)) {
    return []
  }

  return items.map(item => {
    // 权限检查
    const checkPermission = () => {
      if (!Array.isArray(rights) || rights.length === 0) {
        // 没有权限信息时，根据 pagepermisson 决定
        return item.pagepermisson === 1
      }
      // 检查：pagepermisson=1 且 rights 包含该 key
      return item.pagepermisson === 1 && rights.includes(item.key);
    };

    if (!checkPermission()) {
      return null;  // 无权限，不生成菜单项
    }

    // 构建菜单项
    const menuItem = {
      key: item.key,
      label: item.title,
      icon: iconMap[item.key] || null,
      pagepermisson: item.pagepermisson,
      grade: item.grade,
    };

    // 递归处理子菜单
    if (item.children && item.children.length > 0) {
      const filteredChildren = transformMenuData(item.children).filter(Boolean);
      if (filteredChildren.length > 0) {
        menuItem.children = filteredChildren;
      }
    }

    return menuItem;
  }).filter(Boolean);  // 过滤掉 null
}, [rights])
```

**转换过程示例：**
```javascript
// 输入：后端返回的菜单配置
const items = [
  {
    key: "/home",
    title: "首页",
    pagepermisson: 1,
    children: []
  },
  {
    key: "/user-manage",
    title: "用户管理",
    pagepermisson: 1,
    children: [
      { key: "/user-manage/add", title: "添加用户", grade: 2 },
      { key: "/user-manage/list", title: "用户列表", grade: 2 }
    ]
  }
]

// 假设 rights = ["/home", "/user-manage/list"]

// 转换过程：
// item = { key: "/home", ... }
//   → checkPermission() = true (rights 包含 "/home")
//   → 生成 { key: "/home", label: "首页", icon: ..., children: [] }

// item = { key: "/user-manage", ... }
//   → checkPermission() = true (rights 包含 "/user-manage" 或子项)
//   → 处理 children:
//     - "/user-manage/add" → checkPermission() = false (rights 不包含)
//     - "/user-manage/list" → checkPermission() = true (rights 包含)
//   → 生成 { key: "/user-manage", label: "用户管理", children: [{ key: "/user-manage/list", ... }] }

// 最终 menu = [
//   { key: "/home", label: "首页", ... },
//   { key: "/user-manage", label: "用户管理", children: [{ key: "/user-manage/list", ... }] }
// ]
```

## 📚 总结

### 核心流程
1. **登录** → 保存 token（包含 role 对象）
2. **路由守卫** → 解析权限 → 动态生成路由
3. **菜单渲染** → 解析权限 → 过滤菜单 → 渲染菜单树

### 数据流向
```
localStorage.token (JSON字符串)
  ↓
getTokenInfo() 解析
  ↓
token.role.rights (权限数组)
  ↓
路由/菜单权限检查
  ↓
动态生成路由/菜单
```

### 新增角色
- **最小修改**：只需修改 `db.json`，添加角色定义和权限配置
- **业务逻辑**：根据新角色的业务需求，修改相应的业务页面
- **系统优势**：RBAC 权限系统通过配置实现，无需硬编码
